# 页面插入元素

使用 jQuery

添加子标签使用（append 和 prepend）

// 在子标签头部  prepend 或 prependTo

$('.father').prepend('.firstSon')

// 在子标签尾部 append 或 appendTo

$('.father').append('.lastSon')

 添加兄弟元素（after 和 before）

// 添加前面的兄弟 before 或 insertBefore

$('.two').before('.one')

// 添加后面的兄弟 after 或 insertAfter

$('.two').after('.three')

# 屏幕截图（将 html 元素转换成图片进行下载）

场景：

写了一个 div 用来做海报、名片之类的，需要对他进行截取，并下载

<style>
  .box {
    width: 300px;
    height: 300px;
    background-color: red;
  }
</style>

<div class="box"></div>
<button>生成名片</button>

实现：

  1. 引入 html2canvas 插件将 html 元素转换成 canvas

npm i html2canvas

// 返回一个promise，res的参数为canvas对象
html2canvas(document.querySelector('.box'))
1
2
  1. 写一个方法对对 canvas 元素进行下载
function downloadCanvas(canvas) {
  // 调用canvas的toDataUrl方法将canvas变成图片，参数为图片格式，默认为'image/png'
  const canvasUrl = canvas.toDataURL()
  // 生成a标签进行下载
  const oA = document.createElement('a')
  // 设置下载的文件名，默认是'下载'
  oA.download = ''
  oA.href = canvasUrl
  document.body.appendChild(oA)
  oA.click()
  // 下载之后把创建的元素删除
  oA.remove()
}

完整代码

<style>
  .box {
    width: 300px;
    height: 300px;
    background-color: red;

<div class="box"></div>
<button>生成名片</button>

<script>
  document.querySelector('button').addEventListener('click', () => {
    html2canvas(document.querySelector('.box')).then((canvas) => {
      // 调用canvas的toDataUrl方法将canvas变成图片，参数为图片格式，默认为'image/png'
      const canvasUrl = canvas.toDataURL()
      // 生成a标签进行下载
      const oA = document.createElement('a')
      // 设置下载的文件名，默认是'下载'
      oA.download = ''
      oA.href = canvasUrl
      document.body.appendChild(oA)
      oA.click()
      // 下载之后把创建的元素删除
      oA.remove()
    })
  })
</script>

# 虚拟列表

场景： 后台返回十万条或者更多条数据的时候，如果直接渲染到页面上，会极其的慢，用户体验极差，且对浏览器性能影响大，所以可使用虚拟列表 原理： 根据scropTop值，可以判断当前应该显示哪几条数据，从而进行按需渲染 完整代码

<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    .container {
      width: 300px;
      height: 400px;
      border: 1px red solid;
      overflow: auto;
      position: relative
    }

    .box1 {
      width: 100%;
      background-color: yellow;
    }

    .container .list {
      position: absolute;
      top: 0;
      left: 0;
      width: 100%;
      height: 100%;
      z-index: 3;
      transform: translateY(0);
    }

    .container .list div {
      height: 40px;
      width: 100%;
      box-sizing: border-box;
      border: springgreen solid 5px;
    }
  </style>
</head>

<body>
  <div class="container">
    <div class='box1'></div>
    <div class="list"></div>
  </div>
</body>

<script>
  const container = document.querySelector('.container')
  const list = document.querySelector('.list')
  const box1 = document.querySelector('.box1')

  let showList = [] // 实际展示的列表

  const total = 100000
  const allList = []
  for (let i = 1; i <= total; i++) {
    allList.push(i)
  } // 所有数据列表

  // 计算出占位盒子的总高度
  box1.style.height = `${total * 40}px`
  renderList()

  container.addEventListener('scroll', renderList)

  function renderList() {
    list.innerHTML = ''
    // 得出实际展示的盒子里面的元素的下标，结束下标取最大值，避免下面空了一小块
    let start = ~~(container.scrollTop / 40)
    let end = Math.ceil((container.scrollTop + list.offsetHeight) / 40)

    // 将可视区域进行填充
    list.style.transform = `translateY(${start*40}px)`
    showList = allList.slice(start, end)

    showList.forEach((item) => {
      const div = document.createElement('div')
      div.innerText = item
      list.append(div)
    })
  }
</script>

</html>
