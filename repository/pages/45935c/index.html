# 后台表格页面初始代码

<template>
  <div>
    <el-button type="primary" @click="goAdd" size="small">添加</el-button>
    <el-button @click="deleteFn(0)" size="small">批量删除</el-button>
    <el-input
      class="search-input"
      size="small"
      v-model.trim="pageInfo.keyword"
      clearable
      placeholder="请输入"
    >
      <i
        class="el-icon-search el-input__icon"
        slot="suffix"
        @click="getDataList(1)"
      />
    </el-input>
    <el-table :data="dataList">
      <el-table-column
        :prop="item.prop"
        :label="item.label"
        v-for="item in tableConfig"
        :key="item.id"
        :width="item.width"
      >
        <template slot-scope="scope">
          <div v-if="item.prop == 'handle'">
            <el-button type="text" size="small" @click="goEdit(scope.row.id)"
              >修改</el-button
            >
            <el-button type="text" size="small" @click="deleteFn(scope.row.id)"
              >删除</el-button
            >
          </div>
          <div v-else>
            {{ scope.row[item.prop] }}
          </div>
        </template>
      </el-table-column>
    </el-table>
    <el-pagination
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
      :current-page="pageInfo.pageIndex"
      :page-sizes="[10, 25, 50, 100]"
      :page-size="pageInfo.pageSize"
      layout="total, sizes, prev, pager, next, jumper"
      :total="pageInfo.total"
    />
  </div>
</template>

<script>
export default {
  data() {
    return {
      pageInfo: {
        pageIndex: 1,
        pageSize: 10,
        keyword: '',
        total: 0
      },
      dataList: [],
      tableConfig: [
        {
          prop: 'id',
          label: 'ID',
        },
        {
          prop: 'handle',
          label: '操作',
        },
      ],
    }
  },
  created() {
    this.dataList.push({ id: 1 })
    this.getDataList()
  },
  methods: {
    async getDataList() {},
    goAdd() {},
    goEdit() {},
    deleteFn(id) {
      console.log(`删除id为${id}的元素`)
    },
    handleSizeChange(size) {
      this.pageInfo.pageSize = size
      this.getDataList(1)
    },
    handleCurrentChange(pageIndex) {
      this.pageInfo.pageIndex = pageIndex
      this.getDataList()
    },
  },
}
</script>

<template>
  <page-container is-show-breadcrumbs>
    <div class="formList">
      <el-form
        size="small"
        :model="ruleForm"
        :rules="rules"
        ref="ruleForm"
        label-width="100px"
        class="demo-ruleForm"
      >
        <el-form-item label="客户姓名：" prop="customer_name">
          <el-input v-model.trim="ruleForm.customer_name" />
        </el-form-item>
        <el-form-item label="联系电话：" prop="customer_mobile">
          <el-input v-model.trim="ruleForm.customer_mobile" />
        </el-form-item>
        <el-form-item label="微信号：" prop="customer_wx">
          <el-input v-model.trim="ruleForm.customer_wx" />
        </el-form-item>
        <el-form-item label="标签：" prop="customer_tag_id">
          <el-input v-model.trim="ruleForm.customer_tag_id" />
        </el-form-item>
        <el-form-item label="备注：" prop="customer_remark">
          <el-input
            type="textarea"
            :rows="5"
            v-model.trim="ruleForm.customer_remark"
          />
        </el-form-item>
      </el-form>
    </div>
    <div slot="footer">
      <el-button type="primary" size="small" @click="saveInfo('ruleForm')"
        >保存</el-button
      >
      <el-button size="small" @click="cancel">取消</el-button>
    </div>
  </page-container>
</template>

<script>
import PageContainer from '@/components/common/PageContainer.vue'
import { customerApi } from '@/services/api.js'

export default {
  components: {
    PageContainer,
  },
  data() {
    return {
      ruleForm: {
        customer_id: Number(this.$route.params.id),
        customer_name: '',
        customer_mobile: '',
        customer_wx: '',
        customer_tag_id: '',
        customer_remark: '',
      },
      rules: {
        customer_name: [
          {
            type: 'string',
            required: true,
            message: '请输入客户名称',
            trigger: 'blur',
          },
          { min: 1, max: 20, message: '最多输入20个字', trigger: 'blur' },
          {
            pattern: /^[\u4e00-\u9fa5a-zA-Z0-9]+$/,
            message: '名称只能输入中英文数字',
          },
        ],
        customer_mobile: [
          {
            type: 'string',
            required: true,
            message: '请输入11位手机号',
            trigger: 'blur',
          },
          { min: 1, max: 11, message: '请输入11位手机号', trigger: 'blur' },
        ],
      },
    }
  },
  async created() {
    // if (this.ruleForm.customer_id) {
    //   const { data } = await customerApi.findOne();
    //   Object.assign(this.ruleForm, data);
    // }
  },
  methods: {
    async saveInfo(formName) {
      await this.$refs[formName].validate()
      console.log(this.ruleForm)
      debugger
      if (this.ruleForm.customer_id) {
        await customerApi.edit(this.ruleForm)
        this.$message.success('修改成功')
        this.$router.go(-1)
      } else {
        await customerApi.add(this.ruleForm)
        this.$message.success('添加成功')
        this.$router.go(-1)
      }
    },
    cancel() {
      this.$router.go(-1)
    },
  },
}
</script>
